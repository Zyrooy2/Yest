  <script>
    function navegar() {
      let url = document.getElementById("url").value;

      // Si el usuario no pone https://, lo añadimos automáticamente
      if (!url.startsWith("http://") && !url.startsWith("https://")) {
        url = "https://" + url;
      }

      document.getElementById("visor").src = url;
    }
  </script>
